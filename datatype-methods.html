Méthodes Math en JavaScript

JavaScript fournit des méthodes pour manipuler des données numériques, notamment les nombres entiers, les nombres

1. Math.abs(x)

console.log(Math.abs(-10)); // 10 console.log(Math.abs(5)); // 5 console.log(Math.abs(0)); // 0

2. Math.ceil(x)

console.log(Math.ceil(4.2)); // 5 console.log(Math.ceil(7.8)); // 8 console.log(Math.ceil(-2.7)); // -2

3. Math.floor(x)

console.log(Math.floor(4.9)); // 4 console.log(Math.floor(7.1)); // 7 console.log(Math.floor(-2.9)); // -3

4. Math.max(b, c, d)

console.log(Math.max(10, 50, 30)); // 50 console.log(Math.max(10, 50, -30)); // 50

5. Math.min(b, c, d)

console.log(Math.min(10, 50, 30)); // 10 console.log(Math.min(10, 50, -30)); // -30

6. Math.pow(base, exponent)

console.log(Math.pow(2, 3)); // 2^3 = 8 console.log(Math.pow(5, 2)); // 5^2 = 25

7. Math.sqrt(x)

console.log(Math.sqrt(16)); // 4 console.log(Math.sqrt(9)); // 3

8. Math.random() et Math.floor(Math.random() * (max - min + 1)) + min

let randomNum = Math.floor(Math.random() * 10) + 1; console.log(randomNum); // Un nombre entre 1 et 10

9. Math.round(x)

console.log(Math.round(4.4)); // 4 console.log(Math.round(4.6)); // 5

10. Math.trunc(x)

console.log(Math.trunc(4.9)); // 4 console.log(Math.trunc(-7.8)); // -7

11. Math.sign(x)

console.log(Math.sign(10)); // 1 console.log(Math.sign(-5)); // -1 console.log(Math.sign(0)); // 0

12.Math.exp(x)

console.log(Math.exp(1)); // 2.718281828459045 (≈ e^1) console.log(Math.exp(2)); // 7.389056098930649 (≈ e^2)

13. Math.log(x)

console.log(Math.log(1)); // 0 console.log(Math.log(Math.E)); // 1

14. Math.log10(x)

console.log(Math.log10(1000)); // 3 console.log(Math.log10(10)); // 1

15. Math.log2(x)

console.log(Math.log2(8)); // 3 console.log(Math.log2(1024)); // 10

16. Math.cbrt(x)

console.log(Math.cbrt(27)); // 3 console.log(Math.cbrt(8)); // 2

17. Math.hypot(x, y, z)

console.log(Math.hypot(3, 4)); // 5 (car 3² + 4² = 5²) console.log(Math.hypot(6, 8)); // 10

18. Math.imul(a, b)

console.log(Math.imul(5, 6)); // 30 console.log(Math.imul(10, -5)); // -50

19. Math.fround(x)

console.log(Math.fround(4.5)); // 4.5 console.log(Math.fround(4.9)); // 5 console.log(Math.fround(-4.5)); // -4 console.log(Math.fround(-4.9)); // -5

Array

1. push(value)

let myArray = [1, 2, 3]; myArray.push(4); console.log(myArray); // [1, 2, 3, 4]

2. pop()

let myArray = [1, 2, 3, 4]; let poppedValue = myArray.pop(); console.log(myArray); // [1, 2, 3] console.log(poppedValue); // 4

3. shift()

let myArray = [1, 2, 3, 4]; let shiftedValue = myArray.shift(); console.log(myArray); // [2, 3, 4] console.log(shiftedValue); // 1

4. unshift(value)

let myArray = [1, 2, 3]; myArray.unshift(0); console.log(myArray); // [0, 1, 2, 3]

5. concat(value1, value2,...)

let myArray = [1, 2, 3]; let concatArray = myArray.concat([4, 5, 6]); console.log(myArray); // [1, 2, 3] console.log(concatArray); // [1, 2, 3, 4, 5, 6]

6. slice(start, end)

let myArray = [1, 2, 3, 4, 5]; let slicedArray = myArray.slice(1, 4); console.log(myArray); // [1, 2, 3, 4, 5] console.log(slicedArray); // [2, 3, 4]

7. splice(start, deleteCount,...items)

let myArray = [1, 2, 3, 4, 5]; myArray.splice(2, 0, "a", "b", "c"); console.log(myArray); // [1, 2, "a", "b", "c", 3, 4, 5]

8. reverse()

let myArray = [1, 2, 3, 4, 5]; myArray.reverse(); console.log(myArray); // [5, 4, 3, 2, 1]

9. sort(compareFunction)

let myArray = [5, 1, 4, 2, 3]; myArray.sort((a, b) => a - b); console.log(myArray); // [1, 2, 3, 4, 5]

10. indexOf(value, fromIndex)

let myArray = [1, 2, 3, 4, 5]; console.log(myArray.indexOf(3)); // 2 console.log(myArray.indexOf(6)); // -1

11. lastIndexOf(value, fromIndex)

let myArray = [1, 2, 3, 4, 5]; console.log(myArray.lastIndexOf(3)); // 2 console.log(myArray.lastIndexOf(6)); // -1

12. forEach(callbackFn, thisArg)

let myArray = [1, 2, 3, 4, 5]; myArray.forEach((value, index, array) => { console.log(`Index: ${index}, Value: ${value}, Array: ${array}`); });

13. map(callbackFn, thisArg)

let myArray = [1, 2, 3, 4, 5]; let doubledArray = myArray.map(value => value * 2); console.log(myArray); // [1, 2, 3, 4, 5] console.log(doubledArray); // [2, 4, 6, 8, 10]

14. filter(callbackFn, thisArg)

let myArray = [1, 2, 3, 4, 5]; let evenArray = myArray.filter(value => value % 2 === 0); console.log(myArray); // [1, 2, 3, 4, 5] console.log(evenArray); // [2, 4]

15. reduce(callbackFn, initialValue)

let myArray = [1, 2, 3, 4, 5]; let sum = myArray.reduce((accumulator, currentValue) => accumulator + currentValue, 0); console.log(myArray); // [1, 2, 3, 4, 5] console.log(sum); // 15

16. reduceRight(callbackFn, initialValue)

let myArray = [1, 2, 3, 4, 5]; let sum = myArray.reduceRight((accumulator, currentValue) => accumulator + currentValue, 0); console.log(myArray); // [1, 2, 3, 4, 5] console.log(sum); // 15

Object

1. Object.assign(target,...sources)

let obj1 = { a: 1, b: 2 }; let obj2 = { c: 3, d: 4 }; let obj3 = Object.assign({}, obj1, obj2); console.log(obj1); // { a: 1, b: 2 } console.log(obj2); // { c: 3, d: 4 } console.log(obj3); // { a: 1, b: 2, c: 3, d: 4 }

2. Object.create(proto [, propertiesObject])

let obj = Object.create({ x: 10 }); console.log(obj.x); // 10 obj.y = 20; console.log(obj.y); // 20

3. Object.defineProperty(obj, prop, descriptor)

let obj = {}; Object.defineProperty(obj, "property1", { value: 42, writable: false, enumerable: true, configurable: false }); console.log(obj.property1); // 42 obj.property1 = 24; console.log(obj.property1); // 42

4. Object.getOwnPropertyDescriptor(obj, prop)

let obj = { property1: 42, property2: "hello" }; let descriptor = Object.getOwnPropertyDescriptor(obj, "property1"); console.log(descriptor.value); // 42 console.log(descriptor.writable); // false console.log(descriptor.enumerable); // true console.log(descriptor.configurable); // false

5. Object.keys(obj)

let obj = { a: 1, b: 2, c: 3 }; let keys = Object.keys(obj); console.log(keys); // ["a", "b", "c"]

6. Object.values(obj)

let obj = { a: 1, b: 2, c: 3 }; let values = Object.values(obj); console.log(values); // [1, 2, 3]

7. Object.entries(obj)

let obj = { a: 1, b: 2, c: 3 }; let entries = Object.entries(obj); console.log(entries); // [["a", 1], ["b", 2], ["c", 3]]

8. Object.fromEntries(entries)

let entries = [["a", 1], ["b", 2], ["c", 3]]; let obj = Object.fromEntries(entries); console.log(obj); // { a: 1, b: 2, c: 3 }

Match

1. RegExp.prototype.exec(string)

let str = "Hello, World!"; let pattern = /world/i; let result = pattern.exec(str); console.log(result); // ["World", "world", index: 6, input: "Hello, World!"]

2. String.prototype.match(regexp)

let str = "Hello, World!"; let pattern = /world/i; let result = str.match(pattern); console.log(result); // ["World", index: 6, input: "Hello, World!"]

3. String.prototype.search(regexp)

let str = "Hello, World!"; let pattern = /world/i; let result = str.search(pattern); console.log(result); // 6

4. String.prototype.replace(regexp, replacement)

let str = "Hello, World!"; let pattern = /world/i; let result = str.replace(pattern, "Universe"); console.log(result); // Hello, Universe!

5. String.prototype.split(separator, limit)

let str = "Hello, World!"; let pattern = /, /; let result = str.split(pattern, 2); console.log(result); // ["Hello", "World!"]

String

1. String.prototype.charAt(index)

let str = "Hello, World!"; console.log(str.charAt(0)); // H

2. String.prototype.charCodeAt(index)

let str = "Hello, World!"; console.log(str.charCodeAt(0)); // 72

3. String.prototype.concat(...strings)

let str1 = "Hello, "; let str2 = "World!"; let result = str1.concat(str2); console.log(result); // Hello, World!

4. String.prototype.includes(searchString[, position])

let str = "Hello, World!"; console.log(str.includes("World")); // true console.log(str.includes("world")); // true console.log(str.includes("World", 6)); // false

5. String.prototype.indexOf(searchString[, position])

let str = "Hello, World!"; console.log(str.indexOf("World")); // 6 console.log(str.indexOf("world")); // -1 console.log(str.indexOf("World", 7)); // -1

6. String.prototype.lastIndexOf(searchString[, position])

let str = "Hello, World!"; console.log(str.lastIndexOf("World")); // 12 console.log(str.lastIndexOf("world")); // -1 console.log(str.lastIndexOf("World", 11)); // 6

7. String.prototype.slice(beginIndex[, endIndex])

let str = "Hello, World!"; console.log(str.slice(6)); // World! console.log(str.slice(0, 5)); // Hello console.log(str.slice(-3)); // rld

8. String.prototype.substring(beginIndex[, endIndex])

let str = "Hello, World!"; console.log(str.substring(6)); // World! console.log(str.substring(0, 5)); // Hello console.log(str.substring(-3)); // Undef

9. String.prototype.toLowerCase()

let str = "Hello, World!"; console.log(str.toLowerCase()); // hello, world!

10. String.prototype.toUpperCase()

let str = "Hello, World!"; console.log(str.toUpperCase()); // HELLO, WORLD!

11. String.prototype.trimStart()

let str = " Hello, World! "; console.log(str.trimStart()); // Hello, World!

12. String.prototype.trimEnd()

let str = " Hello, World! "; console.log(str.trimEnd()); // Hello, World

13. String.prototype.repeat(count)

let str = "Hello, "; console.log(str.repeat(3)); // Hello, Hello, Hello,

14. String.prototype.padStart(maxLength[, fillString])

let str = "Hello"; console.log(str.padStart(10, "-")); // -----Hello

15. String.prototype.padEnd(maxLength[, fillString])

let str = "Hello"; console.log(str.padEnd(10, "-")); // Hello-----

16. String.prototype.trimLeft()

let str = " Hello, World! "; console.log(str.trimLeft()); // Hello, World!

17. String.prototype.trimRight()

let str = " Hello, World! "; console.log(str.trimRight()); // Hello, World
